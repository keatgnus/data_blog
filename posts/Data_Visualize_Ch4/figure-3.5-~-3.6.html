tx_counties <- US_census %>%
  filter(state == 'Texas') %>%             # Texas 
  select(name, pop2010) %>%                # name,pop2010  
  mutate(county = gsub(' County','',name), # name  County   county 
         popratio = pop2010/median(pop2010)) %>% # pop2010  pop2010    popratio 
  arrange(desc(popratio)) %>%              # popratio   
  mutate(index = 1:n(),
         label = ifelse(index<=3 | index > n()-3 | runif(n()) < .04, county,''))
